731struct dependency_info {
732 const std::type_info& type;
733 int kind;
734 constructor_t defaultConstructor;
735 QString expression; //RESOLVABLE_KIND: The resolvable expression. VALUE_KIND: empty. Otherwise: the required name of the dependency.
736 QVariant value; //VALUE_KIND: The injected value. RESOLVABLE_KIND: the default-value.
737
738 bool has_required_name() const {
739 switch(kind) {
740 case VALUE_KIND:
741 case RESOLVABLE_KIND:
742 return false;
743 default:
744 return !expression.isEmpty();
745
746 }
747 }
748
749};
750
751inline bool operator==(const dependency_info& info1, const dependency_info& info2) {
752 if(info1.kind != info2.kind) {
753 return false;
754 }
755 if(info1.type != info2.type) {
756 return false;
757 }
758 switch(info1.kind) {
759 case VALUE_KIND:
760 return info1.value == info2.value;
761 //In all other cases, we use only the expression. (For RESOLVABLE_KIND, value contains the default-value, which we ignore deliberately)
762 default:
763 return info1.expression == info2.expression;
764 }
