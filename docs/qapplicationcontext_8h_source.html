791struct dependency_info {
792 const std::type_info& type;
793 int kind;
794 constructor_t defaultConstructor;
795 QString expression; //RESOLVABLE_KIND: The resolvable expression. VALUE_KIND: empty. Otherwise: the required name of the dependency.
796 QVariant value; //VALUE_KIND: The injected value. RESOLVABLE_KIND: the default-value.
797
798 bool has_required_name() const {
799 switch(kind) {
800 case VALUE_KIND:
801 case RESOLVABLE_KIND:
802 return false;
803 default:
804 return !expression.isEmpty();
805
806 }
807 }
808
809};
810
811inline bool operator==(const dependency_info& info1, const dependency_info& info2) {
812 if(info1.kind != info2.kind) {
813 return false;
814 }
815 if(info1.type != info2.type) {
816 return false;
817 }
818 switch(info1.kind) {
819 case VALUE_KIND:
820 return info1.value == info2.value;
821 //In all other cases, we use only the expression. (For RESOLVABLE_KIND, value contains the default-value, which we ignore deliberately)
822 default:
823 return info1.expression == info2.expression;
824 }
