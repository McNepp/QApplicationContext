599struct dependency_info {
600 const std::type_info& type;
601 int kind;
602 constructor_t defaultConstructor;
603 QString expression; //RESOLVABLE_KIND: The resolvable expression. VALUE_KIND: empty. Otherwise: the required name of the dependency.
604 QVariant value; //VALUE_KIND: The injected value. RESOLVABLE_KIND: the default-value.
605
606 bool has_required_name() const {
607 switch(kind) {
608 case VALUE_KIND:
609 case RESOLVABLE_KIND:
610 return false;
611 default:
612 return !expression.isEmpty();
613
614 }
615 }
616
617};
618
619inline bool operator==(const dependency_info& info1, const dependency_info& info2) {
620 if(info1.kind != info2.kind) {
621 return false;
622 }
623 if(info1.type != info2.type) {
624 return false;
625 }
626 switch(info1.kind) {
627 case VALUE_KIND:
628 return info1.value == info2.value;
629 //In all other cases, we use only the expression. (For RESOLVABLE_KIND, value contains the default-value, which we ignore deliberately)
630 default:
631 return info1.expression == info2.expression;
632 }
